@Getter @Setter
@Entity
public class Account {

    @Id @GeneratedValue
    private Long id;

    @Column(nullable=false, unique=true)
    private String username;

    private String password;

    @OneToMany(mappedBy = "owner", fetch = FetchType.EAGER) //DEFAULT: LAZY
    private Set<Study> studies = new HashSet<>();

}
